## 각도기

```jsx
// 내가 쓴 답(오답)
function solution(angle){
    if (0 < angle < 90){
        return 1
    } else if (angle = 90){
        return 2
    } else if (90 < angle < 180){
        return 3
    } else if (angle = 180){
        return 4
    }
} 

// 구글링 후 고친 답(통과)
function solution(angle){
    if (0 < angle && angle < 90){
        return 1
    } else if (angle === 90){
        return 2
    } else if (90 < angle && angle < 180){
        return 3
    } else if (angle === 180){
        return 4
    }
}
```
